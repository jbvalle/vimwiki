//Setup
#include <linux/module.h>
#include <linux/init.h>


//Meta information
//Adding License due to some distros not loading free GPL Licensed modules
MODULE_LICENSE("GPL")
MODULE_AUTHOR("NAME") //Edit this
MODULE_DESCRIPTION("Some Description") //Edit this

//Function called when module is loaded to the kernel
static int __init ModuleInit(){
    
    //Output to kernel log
    printk("Hello, Kernel!\n");
    
    //Return succesfull
    return 0;
}

// Function which is called when module is removed from kernel
static void __exit ModuleExit(void){

    printk("Goodbye, KERNEL!\n");
}

//Add function to module initialization
module_init(ModuleInit);
//Add function to module module removal
module_exit(ModuleExit);
